# MISC CLERGY STUFF

namespace = bkt_clergy

# Bishop asks if they can fund construction of a church
bkt_clergy.1001 = {
	type = letter_event
	opening = {
		desc = bkt_clergy.1001.opening
	}
	desc = bkt_clergy.1001.desc
	sender = cp:councillor_court_chaplain
	
	trigger = {
		NOT = { has_character_flag = had_event_bkt_clergy_1001 }
		exists = cp:councillor_court_chaplain
		cp:councillor_court_chaplain = {
			is_physically_able_adult  = yes
			gold >= { 
				value = monumental_gold_value 
				min = main_building_tier_1_cost 
			}
			OR = {
				NOT = { has_trait = greedy }
				gold >= { 
					value = monumental_gold_value
					multiply = 3 
					min = { value = main_building_tier_1_cost multiply = 2 }
				}
			}
		}
		any_held_title = {
			tier = tier_county
			any_county_province = {
				has_holding = no
				has_ongoing_construction = no 
				barony = { barony_cannot_construct_holding = no }
			}
			OR = {
				NOT = {
					any_county_province = {
						has_holding_type = church_holding
					}
				}
				AND = {
					any_county_province = {
						has_holding_type = city_holding
					}
					any_county_province = {
						has_holding_type = castle_holding
					}
					any_county_province = {
						count < 2		#Dont spam 
						has_holding_type = church_holding
					}
				}
			}
		}
	}
	
	immediate = {
		save_scope_as = liege_scope
		cp:councillor_court_chaplain = { save_scope_as = court_chaplain_scope }
		add_character_flag = {
			flag = had_event_bkt_clergy_1001
			years = 15
		}
		add_character_flag = {
			flag = had_any_event_bkt_clergy
			years = 5
		}
		random_held_title = {
			limit = {
				tier = tier_county
				any_county_province = {
					has_holding = no
					has_ongoing_construction = no 
					barony = { barony_cannot_construct_holding = no }
				}
				OR = {
					NOT = {
						any_county_province = {
							has_holding_type = church_holding
						}
					}
					AND = {
						any_county_province = {
							has_holding_type = city_holding
						}
						any_county_province = {
							has_holding_type = castle_holding
						}
						any_county_province = {
							count < 2		#Dont spam 
							has_holding_type = church_holding
						}
					}
				}
			}
			random_county_province = {
				limit = { 
					has_holding = no 
					has_ongoing_construction = no 
					barony = { barony_cannot_construct_holding = no }
				}
				save_scope_as = province_to_construct_at
				barony = { save_scope_as = barony_to_construct_at }	# For loc
			}
		}
		#debug_log = "Clergy event 1001"
		#debug_log_scopes = yes
		#debug_log_date = yes
	}

	option = {		# Accept
		name = bkt_clergy.1001.a
		
		if = {
			limit = { exists = faith.religious_head }
			add_piety = medium_piety_gain
			reverse_add_opinion = {
				target = faith.religious_head
				modifier = pleased_opinion
				opinion = 10
			}
		}
		else = {
			add_piety = major_piety_gain
		}
		
		cp:councillor_court_chaplain = {
			show_as_tooltip = {
				remove_short_term_gold = { 
					value = main_building_tier_1_cost 
					multiply = 0.8
					max = gold
				}
			}
			hidden_effect = {
				remove_short_term_gold = { 
					value = monumental_gold_value 
					min = main_building_tier_1_cost 
					multiply = 0.8
					max = gold
				}
			}
		}
		
		scope:province_to_construct_at = {
			#begin_create_holding = {
			#	type = church_holding
			#	refund_cost = {
			#		gold = 0
			#	}
			#}
			set_holding_type = church_holding
			hidden_effect = {
				create_title_and_vassal_change = {
					type = created
					save_scope_as = change
				}
				barony = {
					change_title_holder_include_vassals = {
						holder = scope:liege_scope
						change = scope:change
					}
				}
				resolve_title_and_vassal_change = scope:change
			}
			add_province_modifier = { 
				modifier = recently_built_holding_modifier 
				years = 5 
			}
			
		}
		
		ai_chance = {
			base = 100
		}
	}

	option = {		# Refuse
		name = bkt_clergy.1001.b
		
		add_piety = major_piety_loss
		
		reverse_add_opinion = {
			target = cp:councillor_court_chaplain
			modifier = irritated_opinion
			opinion = -50
		}
		
		hidden_effect = {
			cp:councillor_court_chaplain = {
				remove_short_term_gold = { 
					value = monumental_gold_value 
					min = main_building_tier_1_cost
					multiply = 0.2
				}
			}
		}
		
		ai_chance = {
			base = 0
		}
	}
}
