every_vassal_has_more_than_value_opinion_trigger = {
	any_vassal = {
		count = all
		OR = {
			highest_held_title_tier = tier_barony	#you dont matter, sorry
			"opinion(liege)" > $VALUE$
		}
	}
}

	NOR = {
		government_has_flag = government_is_landless_adventurer
		government_has_flag = government_is_herder
		government_has_flag = government_is_holy_order
		government_has_flag = government_is_mercenary
		AND = {
			government_has_flag = government_is_celestial
			top_liege = this
		}
		#Covered separatly
		#government_allows = administrative
		#government_allows = deny_powerful_vassal
	}
	OR = {
		has_game_rule = bkt_own_war_declaration_law_toggle_enabled
		AND = {
			is_ai = no
			has_game_rule = bkt_own_war_declaration_law_toggle_player_only
		}
	}
}

can_declare_war_according_to_laws_trigger = {
	trigger_if = {
		limit = { 		#I cheat here for AI counts coz they dont get the opportunity pulse to sway councillors
			is_ai = yes
			highest_held_title_tier <= tier_county
		}
		always = yes
	}
	trigger_if = {
		limit = { 		#Conquerors are cheaty bastards so they can just cheat through the thing.
			has_variable = conqueror
		}
		always = yes
	}
	trigger_else_if = {
		limit = {
			has_realm_law_flag = war_declaration_powerful_vassal_approval
		}
		custom_tooltip = {
			text = no_powerful_vassal_with_negative_opinion_war
			NAND = {
				any_vassal = {
					count > 1
					is_powerful_vassal = yes
					age >= 16
					opposes_succession_law_change_trigger = yes
				}
			}
		}
	}
	trigger_else_if = {
		limit = {
			has_realm_law_flag = war_declaration_council_approval
		}
		custom_tooltip = {
			text = no_real_councillor_with_negative_opinion_war
			NAND = {
				any_councillor = {
					count > 1
					NOR = {
						prev.cp:councillor_spouse ?= this
						prev.cp:councillor_vizier ?= this
					}
					opposes_succession_law_change_trigger = yes
				}
			}
		}
	}
	trigger_else_if = {
		limit = {
			has_realm_law_flag = war_declaration_no_approval
		}
		always = yes
	}
	trigger_else = {	#just in case
		always = yes
	}
}
