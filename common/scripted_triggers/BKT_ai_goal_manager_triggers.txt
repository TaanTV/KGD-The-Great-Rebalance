}

		save_temporary_scope_as = story_owner_scope

story_owner_scope_can_attack_this = {
	NOR = {
		this = scope:story_owner_scope 
		any_liege_or_above = { this = scope:story_owner_scope } 
		is_tributary_of_suzerain_or_above = scope:story_owner_scope
		scope:story_owner_scope = { is_tributary_of_suzerain_or_above = PREV }
		scope:story_owner_scope.liege ?= this
		scope:story_owner_scope.liege.liege ?= this
		is_allied_to = scope:story_owner_scope
		scope:story_owner_scope = { has_truce = PREV }
		has_hook = scope:story_owner_scope 
		reverse_opinion = {
			target = scope:story_owner_scope 
			value >= 50
		}
	}
	#max_military_strength <= { value = scope:story_owner_scope.max_military_strength multiply = 1.5 }
}
