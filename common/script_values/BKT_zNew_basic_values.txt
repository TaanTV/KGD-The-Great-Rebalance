average_vassal_opinion = {
	value = 0
	if = {
		limit = {
			is_landed = yes
			any_vassal = { exists = this highest_held_title_tier > tier_barony }
		}
		every_vassal = {
			limit = { highest_held_title_tier > tier_barony }
			add = "opinion(liege)"
		}
		divide = {
			value = {
				every_vassal = {
					limit = { highest_held_title_tier > tier_barony }
					add = 1
				}
			}
			min = 1
		}	
		floor = yes
	}
	else = {
		value = 100
	}
}

own_war_declaration_law_prestige_cost = {
	if = {
		limit = {
			OR = {
				has_realm_law_in_group = crown_authority 
				government_allows = administrative
			}
		}
		add = increase_crown_authority_prestige_cost
	}
	else_if = {
		limit = {
			OR = {
				has_realm_law_in_group = tribal_authority 
				has_realm_law_in_group = nomadic_authority 
			}
		}
		add = increase_tribal_authority_prestige_cost
	}
	else = {
		add = 1000
	}
}

#tiny_gold_max_value_static = 50
tiny_gold_max_less_scaled_value = {
	value = tiny_gold_max_value_static
	if = {
		limit = {
			exists = culture
		}
		culture = {
			if = {
				limit = {
					has_cultural_era_or_later = culture_era_late_medieval
				}
				multiply = 2
			}
			else_if = {
				limit = {
					has_cultural_era_or_later = culture_era_high_medieval
				}
				multiply = 1.5
			}
			else_if = {
				limit = {
					has_cultural_era_or_later = culture_era_early_medieval
				}
				multiply = 1
			}
			else = {
				multiply = 0.75
			}
		}
	}
}
