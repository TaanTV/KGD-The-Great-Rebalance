	value = 25		#prev 30

	if = { #Unop: There are inconsistency in calls with this, sometimes logs throw errors stating root is an activity where it should be a character, also sometimes called from interaction where root does not exists
		limit = { # Root is a character
			exists = root.primary_title
		}
		multiply = {
			value = root.primary_title.tier
			subtract = 1
			min = 1
			#ADDED
			#multiply = 0.75		# Scale less aggresively with tier
		}
		if = {
			limit = {
				root = {
					government_has_flag = government_is_tribal
				}
			}
			multiply = 0.5
		}
		if = {
			limit = {
				root.primary_title.tier > tier_duchy
			}
			#add = 50

			add = { value = root.primary_title.tier add = -3 multiply = 20 }
		}

			limit = {
				is_ai = yes

			}

		}

			limit = {
				is_ai = yes

			}

			multiply = 0.5
		}

			limit = {
				root.activity_host.primary_title.tier > tier_duchy
			}
			#add = 50
			#ADDED - Wtf? not like that.
			add = { value = root.activity_host.primary_title.tier add = -3 multiply = 20 }
		}
		if = {
			limit = {
				root.activity_host = { is_ai = yes }
				has_game_rule = hard_difficulty
			}
			multiply = 0.5
		}
		else_if = {
			limit = {
				root.activity_host = { is_ai = yes }
				has_game_rule = very_hard_difficulty
			}
			multiply = 0.25
		}
	}
	else_if = {

			exists = scope:actor
		}
		multiply = {
			value = scope:actor.primary_title.tier

			min = 1

activity_cost_scale_by_era = {
	value = 0
	if = {
		limit = {
			exists = culture
		}
		culture = {
			if = {
				limit = {
					has_cultural_era_or_later = culture_era_late_medieval
				}
				add = 2.25		#prev 3
				# What? 
			}
			else_if = {
				limit = {
					has_cultural_era_or_later = culture_era_high_medieval
				}
				add = 1.5
			}
			else_if = {
				limit = {
					has_cultural_era_or_later = culture_era_early_medieval
				}
				add = 1
			}
			else = {
				add = 0.75
			}
		}
	}
	else_if = { #Unop: To handle cases where 'root' is an activity
		limit = { exists = activity_host }
		if = {
			limit = { exists = activity_host.culture }
			activity_host.culture = {
				if = {
					limit = { has_cultural_era_or_later = culture_era_late_medieval }
					add = 2.25		#prev 3
				}
				else_if = {
					limit = { has_cultural_era_or_later = culture_era_high_medieval }
					add = 1.5
				}
				else_if = {
					limit = { has_cultural_era_or_later = culture_era_early_medieval }
					add = 1
				}
				else = {
					add = 0.75
				}
			}
		}
	}
}
