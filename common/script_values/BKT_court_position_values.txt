major_court_position_prestige_revoke_cost = {
	value = 50		#prev 200
	if = {
		limit = {
			exists = this
			has_character_flag = temporary_court_position_cost_removal
		}
		multiply = 0
	}
	#ADDED
	else_if = {
		limit = {
			exists = this
			is_ruler = yes		#this is redundant. I hope.
			highest_held_title_tier >= tier_duchy
		}
		multiply = {
			value = -2
			add = highest_held_title_tier
			multiply = 2
			max = 7				#because 350 is a funny prestige number.
		}
	}
}
