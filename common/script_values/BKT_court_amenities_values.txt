	value = sub_realm_size
	#ADDED all of the below
	multiply = 2
	if = {
		limit = {	#If tribal - only count in the capital and triple the sub_realm_size
			government_has_flag = government_is_tribal
			government_has_flag = government_is_nomadic
		}
		multiply = 3
		add = {
			value = capital_county.development_level
			multiply = 0.5
		}
	}
	else = {		#Else - scale up with total dev
		add = {
			every_sub_realm_county = {
				add = development_level
			}
			multiply = 0.1
		}		
		add = {		# Count domain twice
			every_held_title = {
				limit = { tier = tier_county }
				add = development_level
			}
			multiply = 0.1
		}
	}
	#ADDED - that divide is actually non-vanilla
	divide = 3		#prev 4
	#END of ADDED
	divide = {
		add = 4
		subtract = {
			add = 0.25
			multiply = court_grandeur_current_level
		}
	}
	if = { # Add a baseline depending on rank
		limit = {
			highest_held_title_tier = tier_empire
			#ADDED - replaced
			highest_held_title_tier >= tier_empire
		}
		add = 40		#20
	}
	else = {
		add = 20		#10
	}
	if = {
		limit = {
			culture = {
				has_cultural_era_or_later = culture_era_late_medieval
			}
		}
		multiply =  {
			value =  1.75
		}
	}
	else_if = {
		limit = {
			culture = {
				has_cultural_era_or_later = culture_era_high_medieval
			}
		}
		multiply = {
			value = 1.5
		}
	}
	else_if = {
		limit = {
			culture = {
				has_cultural_era_or_later = culture_era_early_medieval
			}
		}
		multiply = {
			value = 1.25
		} 
	}
	if = {
		limit = {
			has_game_rule = hard_difficulty
			is_ai = yes
		}
		multiply = 0.75
	}
	else_if = {
		limit = {
			has_game_rule = very_hard_difficulty
			is_ai = yes
		}
		multiply = 0.5
	}
	#ADDED
	if = {
		limit = { government_has_flag = government_has_higher_grandeur_penalty_and_court_costs }
		multiply = 1.5
	}
}
