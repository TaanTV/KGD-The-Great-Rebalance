knight_prestige_gain_on_wound = {
	value = minor_prestige_gain
	if = {
		limit = {
			exists = scope:enemy_knight.primary_title
			scope:enemy_knight.primary_title.tier > 0
		}
		#multiply = scope:enemy_knight.primary_title.tier
		#ADDED - barons are people too
		multiply = {
			value = scope:enemy_knight.primary_title.tier
			add = 1
		}
	}
	if = {
		limit = {
			exists = scope:enemy_knight
			scope:enemy_knight = {
				is_lowborn = yes
			}
		}
		#divide = 2
		#ADDED - lowborns can get a bit more prestige
		if = {
			limit = { is_lowborn  = yes }
			divide = 1.5
		}
		else = { divide = 2 }
	}
}

#Send Gift
send_gift_opinion = {
	value = 15	#prev 5
	add = {
		if = {
			limit = {
				scope:actor.diplomacy > 0
			}
			add = {
				value = scope:actor.diplomacy
				multiply = 2	#prev 3
			}
		}
		if = {
			limit = {
				scope:recipient.ai_greed > 0
			}
			add = {
				value = scope:recipient.ai_greed
				divide = 4		#prev 3
			}
		}
		if = {
			limit = {
				scope:recipient.liege = scope:actor
			}
			divide = 1.5
		}
		if = {
			limit = {
				scope:recipient = scope:actor.faith.religious_head
			}
			divide = 3
		}
		if = {
			limit = {
				scope:actor.culture ?= {
					has_innovation = fp3_innovation_fritware
				}
			}
			multiply = 1.2
		}
		max = 85	#prev 100
	}
	if = {
		limit = {
			scope:actor.culture = { has_cultural_parameter = more_gift_opinion }
		}
		multiply = 1.2
	}
	if = {
		limit = {
			scope:actor = {
				any_sub_realm_barony = {
					county.holder = scope:actor
					title_province = { 
						has_building_or_higher = dragon_kiln_01
					}
				}
			}
		}
		multiply = 1.5
	}
	if = {
		limit = {
			scope:actor = { has_perk = thoughtful_perk }
		}
		multiply = 1.5		#prev 2
	}
}

#FOR DIVORCE
divorce_cost = {
	value = 750		#prev 100
	if = {
		limit = {
			exists = scope:actor
			NOT = { exists = scope:secondary_actor }	#Dynast interactions can go f themselves.
			OR = {
				scope:recipient ?= {
					is_spouse_of = scope:actor
					NOT = { exists = scope:secondary_recipient }
					OR = {
						reverse_has_opinion_modifier = {
							modifier = unfaithful_spouse_exposed_opinion
							target = scope:actor
						}
						reverse_has_opinion_modifier = {
							modifier = unfaithful_spouse_discovered_opinion
							target = scope:actor
						}
					}
				}
				scope:secondary_recipient ?= {
					is_spouse_of = scope:actor
					OR = {
						reverse_has_opinion_modifier = {
							modifier = unfaithful_spouse_exposed_opinion
							target = scope:actor
						}
						reverse_has_opinion_modifier = {
							modifier = unfaithful_spouse_discovered_opinion
							target = scope:actor
						}
					}
				}
			}
		}
		add = -700
	}
}
