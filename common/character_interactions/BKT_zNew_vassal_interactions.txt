# Request a council position from your liege in exchange for influence
request_council_powerful_vassal_interaction = {
	category = interaction_category_vassal
	interface_priority = 119 #listed right after force_onto_council
	icon = council

	desc = {
		first_valid = {
			triggered_desc = {
				trigger = { scope:actor = { is_powerful_vassal = yes } }
				desc = request_council_powerful_vassal_interaction_desc
			}
			desc = request_council_powerful_vassal_interaction_desc_non_powerful
		}
	}

	special_interaction = request_council_position
	interface = council_task_interaction
	ai_maybe = yes
	ai_min_reply_days = 4
	ai_max_reply_days = 9
	can_send_despite_rejection = yes
	popup_on_receive = yes
	pause_on_receive = yes
	send_name = "SEND_REQUEST_COUNCIL"

	greeting = positive
	notification_text = REQUEST_COUNCIL_PROPOSAL

	pre_answer_yes_key = ACCEPT
	pre_answer_no_key = DECLINE

	#on_decline_summary = request_council_decline_summary

	ai_targets = {
		ai_recipients = liege
	}
	ai_target_quick_trigger = {
		adult = yes
	}
	ai_frequency_by_tier = {
		barony = 120
		county = 36
		duchy = 24
		kingdom = 12
		empire = 12
		hegemony = 0
	}

	cooldown_against_recipient = { years = 10 }

	cost = {
		prestige = {
			if = {
				limit = { scope:actor = { is_powerful_vassal = yes } }
				value = medium_prestige_value
			}
			else = {
				value = major_prestige_value
			}
			if = {
				limit = { scope:recipient = { has_realm_law = crown_authority_0 } }
				multiply = 0.5
			}
		}
	}

	is_shown = {
		scope:recipient != scope:actor
		scope:actor = {
			liege ?= scope:recipient
			is_powerful_vassal = yes
		}
		#No tribes, i think.... Actually, allow tribes, but only vs feudals.
		scope:actor = { 
			OR = {
				government_has_flag = government_is_feudal
				government_has_flag = government_is_republic
				government_has_flag = government_is_clan
				government_has_flag = government_is_tribal
			}
		}
		scope:recipient = { 
			OR = {
				government_has_flag = government_is_feudal
				government_has_flag = government_is_clan
			}
		}
		trigger_if = {
			limit = {
				exists = scope:target.councillor
			}
			scope:target = { can_fire_position = yes }
		}
	}

	is_valid_showing_failures_only = {
		scope:actor = {
			is_available_adult = yes
			NOT = {
				is_at_war_with = scope:recipient
			}
			NOT = {
				is_councillor = yes
			}
			NAND = {
				has_council_position = councillor_court_chaplain
				faith = {
					OR = {
						has_doctrine = doctrine_clerical_succession_temporal_fixed_appointment
						has_doctrine = doctrine_clerical_succession_spiritual_fixed_appointment
					}
				}
			}
		}
		scope:recipient = {
			custom_tooltip = {
				text = request_council_powerful_vassal_interaction_cooldown_tt
				NOT = { has_variable = request_council_powerful_vassal_interaction_cooldown }
			}
		}
	}

	has_valid_target_showing_failures_only = {
		trigger_if = {
			limit = {
				exists = scope:target.councillor
			}
			scope:target.councillor = {
				NOT = {
					is_blocked_from_being_fired_from_council_trigger = {
						COURT_OWNER = scope:recipient
					}
				}
			}
		}
	}

	on_accept = {
		scope:target = {
			save_scope_as = council_task
		}
		scope:actor = {
			#block_firing_councillor_effect = { COURT_OWNER = scope:recipient }
			set_variable = {
				name = block_fire_councillor
				value = scope:recipient
				years = council_position_lite_locked_value
			}
			custom_tooltip = request_council.recipient.tt.duration
			#And notify exactly which councillor type they've become.
			hidden_effect = {
				send_interface_toast = {
					type = event_toast_effect_good
					title = force_onto_council.actor.tt
					left_icon = scope:recipient
					custom_tooltip = request_council.recipient.tt.duration
					#show_as_tooltip = {
					#	change_influence = admin_council_loss_value
					#}
				}
			}
			add_opinion = {
				target = scope:recipient
				modifier = pleased_opinion
				opinion = 15
			}
			stress_impact = { content = minor_stress_impact_gain }
		}
		scope:recipient = {
			assign_council_task = {
				target = scope:actor
				council_task = scope:council_task
			}
			custom_tooltip = force_onto_council.desc.tt
			#And notify exactly which councillor type they've become.
			hidden_effect = {
				send_interface_toast = {
					type = event_toast_effect_neutral
					title = force_onto_council.recipient.tt
					left_icon = scope:actor
					custom_tooltip = request_council.recipient.tt.duration
				}
				set_variable = {
					name = request_council_powerful_vassal_interaction_cooldown
					years = 1
				}
			}
			if = {
				limit = { is_ai = no NOT = { has_trait = arbitrary } }
				stress_impact = {
					greedy = miniscule_stress_impact_gain
					arrogant = miniscule_stress_impact_gain
				}
			}
			else = {
				#custom_tooltip = request_council_powerful_vassal_interaction.stress_impact.arbitrary
			}
		}
	}

	on_decline = {
		scope:actor = {
			progress_towards_rival_effect = {
				REASON = rival_grave_insult
				CHARACTER = scope:recipient
				OPINION = -30
			}
			if = {
				limit = { is_powerful_vassal_of = scope:recipient }
				scope:recipient = {
					every_powerful_vassal = {
						custom = custom.every_powerful_vassal
						limit = { this != scope:actor }
						add_opinion = {
							target = scope:recipient
							modifier = disappointed_opinion
							opinion = -15
						}
					}
				}
			}
			#And notify the vassal.
			hidden_effect = {
				send_interface_toast = {
					type = event_toast_effect_neutral
					title = bkt_request_council.refused.actor.tt
					left_icon = scope:actor
					show_as_tooltip = {
						scope:recipient = {
							if = {
								limit = { 
									has_legitimacy = yes 
									NOT = { has_realm_law = crown_authority_3 }
								}
								if = {
									limit = { 
										highest_held_title_tier >= tier_empire 
										scope:actor = { is_powerful_vassal = yes } 
									}
									add_legitimacy = major_legitimacy_loss
								}
								else_if = {
									limit = { 
										OR = {
											highest_held_title_tier >= tier_empire 
											scope:actor = { is_powerful_vassal = yes } 
										}
									}
									add_legitimacy = medium_legitimacy_loss
								}
								else = {
									add_legitimacy = minor_legitimacy_loss
								}
							}
							else_if = {
								limit = { has_legitimacy = yes }
								add_legitimacy = miniscule_legitimacy_loss
							}
						}
					}
				}
			}
		}
		scope:recipient = {
			if = {
				limit = { 
					has_legitimacy = yes 
					NOT = { has_realm_law = crown_authority_3 }
				}
				if = {
					limit = { 
						highest_held_title_tier >= tier_empire 
						scope:actor = { is_powerful_vassal = yes } 
					}
					add_legitimacy = major_legitimacy_loss
				}
				else_if = {
					limit = { 
						OR = {
							highest_held_title_tier >= tier_empire 
							scope:actor = { is_powerful_vassal = yes } 
						}
					}
					add_legitimacy = medium_legitimacy_loss
				}
				else = {
					add_legitimacy = minor_legitimacy_loss
				}
			}
			else_if = {
				limit = { has_legitimacy = yes }
				add_legitimacy = miniscule_legitimacy_loss
			}
			add_character_flag = {
				flag = refused_council_request
				years = 5
			}
			set_variable = {
				name = request_council_powerful_vassal_interaction_cooldown
				years = 1
			}
			if = {
				limit = { is_ai = no NOT = { has_trait = arbitrary } }
				stress_impact = {
					generous = miniscule_stress_impact_gain
					just = minor_stress_impact_gain
					humble = miniscule_stress_impact_gain
				}
			}
			else = {
				#custom_tooltip = request_council_powerful_vassal_interaction.stress_impact.arbitrary
			}
		}
	}

	ai_set_target = {
		scope:actor = {
			random_list = {
				10 = {
					trigger = {
						exists = scope:recipient.council_task:councillor_chancellor
						can_be_chancellor_trigger = { COURT_OWNER = scope:recipient }
					}
					modifier = {
						add = diplomacy
					}
					modifier = {
						trigger = { highest_skill = diplomacy }
						factor = 10
					}
					modifier = {
						trigger = { diplomacy < 8 }
						factor = 0.5
					}
					scope:recipient.council_task:councillor_chancellor = {
						save_scope_as = target
					}
				}
				10 = {
					trigger = {
						exists = scope:recipient.council_task:councillor_steward
						can_be_steward_trigger = { COURT_OWNER = scope:recipient }
					}
					modifier = {
						add = stewardship
					}
					modifier = {
						trigger = { highest_skill = stewardship }
						factor = 10
					}
					modifier = {
						trigger = { stewardship < 8 }
						factor = 0.5
					}
					scope:recipient.council_task:councillor_steward = {
						save_scope_as = target
					}
				}
				10 = {
					trigger = {
						exists = scope:recipient.council_task:councillor_marshal
						can_be_marshal_trigger = { COURT_OWNER = scope:recipient }
					}
					modifier = {
						add = martial
					}
					modifier = {
						trigger = { highest_skill = martial }
						factor = 10
					}
					modifier = {
						trigger = { martial < 8 }
						factor = 0.5
					}
					scope:recipient.council_task:councillor_marshal = {
						save_scope_as = target
					}
				}
				10 = {
					trigger = {
						exists = scope:recipient.council_task:councillor_spymaster
						can_be_spymaster_trigger = { COURT_OWNER = scope:recipient }
					}
					modifier = {
						add = intrigue
					}
					modifier = {
						trigger = { highest_skill = intrigue }
						factor = 10
					}
					scope:recipient.council_task:councillor_spymaster = {
						save_scope_as = target
					}
				}
				10 = {
					trigger = {
						exists = scope:recipient.council_task:councillor_court_chaplain
						can_be_court_chaplain_trigger = { COURT_OWNER = scope:recipient }
					}
					modifier = {
						add = learning
					}
					modifier = {
						trigger = { highest_skill = learning }
						factor = 10
					}
					modifier = {
						trigger = { learning < 8 }
						factor = 0.5
					}
					scope:recipient.council_task:councillor_court_chaplain = {
						save_scope_as = target
					}
				}
			}
		}
	}

	ai_potential = {
		top_liege != this
		is_councillor = no
		liege ?= {
			NOT = { has_character_flag = refused_council_request }
			is_at_war = no
		}
	}

	ai_will_do = {
		base = 5

		# Increase chance if greed and/or boldness is high enough
		modifier = {
			add = 20
			OR = {
				ai_greed >= medium_positive_ai_value
				ai_boldness >= medium_positive_ai_value
			}
		}
		modifier = {
			factor = 2
			ai_greed >= very_high_positive_ai_value
		}
		modifier = {
			factor = 2
			ai_boldness >= very_high_positive_ai_value
		}
		modifier = {
			factor = 1.5
			liege.highest_held_title_tier = tier_kingdom
		}
		modifier = {
			factor = 2
			liege.highest_held_title_tier >= tier_empire
		}

		# Don't do it towards a good relation, unless very greedy
		modifier = {
			factor = 0
			scope:actor = {
				ai_greed < very_high_positive_ai_value
				OR = {
					has_relation_friend = scope:recipient
					has_relation_lover = scope:recipient
					has_relation_best_friend = scope:recipient
					has_relation_soulmate = scope:recipient
					"scope:actor.opinion(scope:recipient)" > 75
				}
			}
		}

		# No need if they are a councillor already
		modifier = {
			factor = 0
			is_councillor = yes
		}
	}

	ai_accept = {
		base = 0
		modifier = {
			add = -50
			desc = BASE_RELUCTANCE
		}
		opinion_modifier = { # Opinion of you
			who = scope:recipient
			opinion_target = scope:actor
			multiplier = 0.75
			desc = AI_OPINION_REASON
		}
		opinion_modifier = { # Opinion of the current councillor
			trigger = {
				exists = scope:target.councillor
			}
			who = scope:recipient
			opinion_target = scope:target.councillor
			multiplier = -0.75
			desc = AI_OPINION_REASON
		}
		ai_value_modifier = {
			ai_energy = 0.25 		#It's a change
			ai_boldness = -0.25		#Cowards be cowards
			ai_honor = {
				value = 0
				if = {
					limit = { scope:actor = { is_powerful_vassal = yes } }
					value = 0.35	#Powerful vassals have the rights to be on the council
				}
			}
		}
		#Relations
		modifier = {
			scope:actor = {
				has_relation_rival = scope:recipient
			}
			add = -50
			desc = AI_YOUR_RIVAL
		}
		modifier = {
			scope:actor = {
				has_relation_friend = scope:recipient
			}
			add = 25
			desc = AI_YOUR_FRIEND
		}
		#You are better at the job
		modifier = {
			scope:target = council_task:councillor_chancellor
			scope:actor.diplomacy > scope:target.councillor.diplomacy
			desc = AI_BETTER_AT_THE_JOB
			add = {
				value = scope:actor.diplomacy
				subtract = scope:target.councillor.diplomacy
				multiply = 5
				if = {
					limit = { scope:actor = { is_ai = yes } }
					multiply = 2
				}
			}
		}
		modifier = {
			scope:target = council_task:councillor_marshal
			scope:actor.martial > scope:target.councillor.martial
			desc = AI_BETTER_AT_THE_JOB
			add = {
				value = scope:actor.martial
				subtract = scope:target.councillor.martial
				multiply = 5
				if = {
					limit = { scope:actor = { is_ai = yes } }
					multiply = 2
				}
			}
		}
		modifier = {
			scope:target = council_task:councillor_steward
			scope:actor.stewardship > scope:target.councillor.stewardship
			desc = AI_BETTER_AT_THE_JOB
			add = {
				value = scope:actor.stewardship
				subtract = scope:target.councillor.stewardship
				multiply = 5
				if = {
					limit = { scope:actor = { is_ai = yes } }
					multiply = 2
				}
			}
		}
		modifier = {
			scope:target = council_task:councillor_spymaster
			scope:actor.intrigue > scope:target.councillor.intrigue
			desc = AI_BETTER_AT_THE_JOB
			add = {
				value = scope:actor.intrigue
				subtract = scope:target.councillor.intrigue
				multiply = 5
				if = {
					limit = { scope:actor = { is_ai = yes } }
					multiply = 2
				}
			}
		}
		modifier = {
			scope:target = council_task:councillor_court_chaplain
			scope:actor.learning > scope:target.councillor.learning
			desc = AI_BETTER_AT_THE_JOB
			add = {
				value = scope:actor.learning
				subtract = scope:target.councillor.learning
				multiply = 5
				if = {
					limit = { scope:actor = { is_ai = yes } }
					multiply = 2
				}
			}
		}
		#You are significantly WORSE at the job (we start reducing AI acceptance when it is more than 5 below current councillor's skill)
		modifier = {
			scope:target = council_task:councillor_chancellor
			scope:actor.diplomacy < { value = scope:target.councillor.diplomacy subtract = 5 }
			desc = AI_WORSE_AT_THE_JOB
			add = {
				value = scope:actor.diplomacy
				subtract = scope:target.councillor.diplomacy
				add = 5
				multiply = 2
			}
		}
		modifier = {
			scope:target = council_task:councillor_marshal
			scope:actor.martial < { value = scope:target.councillor.martial subtract = 5 }
			desc = AI_WORSE_AT_THE_JOB
			add = {
				value = scope:actor.martial
				subtract = scope:target.councillor.martial
				add = 5
				multiply = 2
			}
		}
		modifier = {
			scope:target = council_task:councillor_steward
			scope:actor.stewardship < { value = scope:target.councillor.stewardship subtract = 5 }
			desc = AI_WORSE_AT_THE_JOB
			add = {
				value = scope:actor.stewardship
				subtract = scope:target.councillor.stewardship
				add = 5
				multiply = 2
			}
		}
		modifier = {
			scope:target = council_task:councillor_spymaster
			scope:actor.intrigue < { value = scope:target.councillor.intrigue subtract = 5 }
			desc = AI_WORSE_AT_THE_JOB
			add = {
				value = scope:actor.intrigue
				subtract = scope:target.councillor.intrigue
				add = 5
				multiply = 2
			}
		}
		modifier = {
			scope:target = council_task:councillor_court_chaplain
			scope:actor.learning < { value = scope:target.councillor.learning subtract = 5 }
			desc = AI_WORSE_AT_THE_JOB
			add = {
				value = scope:actor.learning
				subtract = scope:target.councillor.learning
				add = 5
				multiply = 2
			}
		}
		#You are more powerful
		modifier = {
			scope:actor = { is_powerful_vassal = yes }
			OR = {
				NOT = { exists = scope:target.councillor }
				scope:target.councillor = { is_powerful_vassal = no }
			}
			desc = AI_POWERFUL_VASSAL
			add = 50
		}
	}

				skills = { stewardship intrigue } 
				target = scope:recipient
				# Victory! Gold for scope:actor, maluses for the liege.
				50 = {
					compare_modifier = {
						value = scope:duel_value
						multiplier = 2
						min = -49
					}
					# The bigger the liege the easier it is to do
					modifier = {
						add = {
							value = scope:recipient.highest_held_title_tier
							add = -3
							multiply = 5
						}
						always = yes
					}
					# Bonus for spouses (yes they intentionally stack)
					modifier = {
						add = 30
						scope:recipient = {
							is_married = yes
							primary_spouse = scope:actor
						}
					}
					modifier = {
						add = 20
						scope:recipient = {
							is_married = yes
							any_spouse = { this = scope:actor }
						}
					}
					modifier = {
						add = 15
						scope:recipient = {
							is_imprisoned = yes
						}
					}
					modifier = {
						add = 5
						scope:recipient = {
							is_travelling = yes
						}
					}
					# You always have some kind of a chance.
					min = 10
					desc = diarch_syphon_treasury_interaction.tt.success
					# Inform scope:actor what type of embezzlement they'll be pursuing.
					## Very high stewardship characters syphon development growth from a few counties.
					custom_tooltip = vassal_syphon_treasury_interaction.tt.success.syphon_control_several_counties 
					
					save_temporary_scope_value_as = {
						name = gold_to_get
						value = {
							add = scope:actor.minor_gold_value
							multiply = 0.5
							add = {
								value = scope:recipient.minor_gold_value
								multiply = 0.5
							}
							multiply = 0.2
							floor = yes
							multiply = 5
						}
					}
					
					# Now we sort the actual effects.
					## A nice toast to tell us about our new gold.
					send_interface_toast = {
						type = event_toast_effect_good
						title = diarch_syphon_treasury_interaction.tt.success
						left_icon = scope:recipient
						# Transfer gold.
						## Small amount.
						give_or_update_embezzler_secret_effect = {
							TARGET = scope:recipient
							STAKE = scope:gold_to_get	#prev syphon_treasury_gold_gain_low
						}
						add_gold = scope:gold_to_get	#syphon_treasury_gold_gain_low
					}
					## Plus the maluses for the liege.
					scope:recipient = {
						trigger_event = {
							days = { 1 9 }
							on_action = on_success_vassal_syphon_treasury_interaction
						}
						set_variable = {
							name = vassal_syphon_treasury_interaction_cooldown
							years = 1
						}
					}
				}
				# Defeat! No gold, just prestige loss & a crime opinion.
				50 = {
					compare_modifier = {
						value = scope:duel_value
						multiplier = -3
						min = -49
					}
					# No way you're stealing from paranoics
					modifier = {
						add = -15
						scope:recipient = { has_trait = paranoid }
					}
					# Or greedy bastards
					modifier = {
						add = -10
						scope:recipient = { has_trait = greedy }
					}
					# You can always fail.
					min = 20
					desc = diarch_syphon_treasury_interaction.tt.failure.actor
					send_interface_toast = {
						type = event_toast_effect_bad
						title = diarch_syphon_treasury_interaction.tt.failure.actor
						left_icon = scope:recipient
						show_as_tooltip = {
							scope:actor = {
								add_prestige = medium_prestige_loss
								reverse_add_opinion = {
									target = scope:recipient
									modifier = embezzled_opinion
								}
							}
						}
					}
					hidden_effect = {
						scope:recipient = {
							send_interface_toast = {
								type = event_toast_effect_neutral
								title = diarch_syphon_treasury_interaction.tt.failure.recipient
								left_icon = scope:actor
								scope:actor = {
									add_prestige = medium_prestige_loss
									reverse_add_opinion = {
										target = scope:recipient
										modifier = embezzled_opinion
									}
								}
							}
						}
					}
				}
			}
			if = {
				limit = { is_ai = no }
				stress_impact = {
					generous = major_stress_impact_gain
					honest = major_stress_impact_gain
					just = medium_stress_impact_gain
					compassionate = medium_stress_impact_gain
				}
			}
		}

		# Debug logging.
		if = {
			limit = {
				always = no
			}
			debug_log = "AI used vassal_syphon_treasury_interaction!"
			save_temporary_scope_value_as = {
				name = liege_stats
				value = { add = scope:recipient.stewardship add = scope:recipient.intrigue }
			}
			save_temporary_scope_value_as = {
				name = vassal_stats
				value = { add = scope:actor.stewardship add = scope:actor.intrigue }
			}
			save_temporary_scope_value_as = {
				name = vassal_opinion_of_liege
				value = "scope:actor.opinion(scope:recipient)"
			}
			debug_log_scopes = yes
		}
	}

		#Should there be more? Idk...
