### High Almoner Tasks ###

high_almoner_attend_poor = {
	court_position_types = {  high_almoner_court_position chief_qadi_court_position court_brahmin_court_position }

	# Monthly cost for the task
	cost = {
		round = no
		gold = {
			value = 0
			if = {
				limit = {
					scope:liege = {
						has_treasury = no
					}
				}
				add = {
					value = monthly_court_position_task_cost
					desc = COURT_POSITION_TASK_COST_BREAKDOWN_BASE
					format = "BASE_VALUE_FORMAT_DECIMALS_PLUS_NEGATIVE"
				}
			}
		}
		treasury = {
			value = 0
			if = {
				limit = {
					scope:liege = {
						has_treasury = yes
					}
				}
				add = {
					value = monthly_court_position_task_cost
					desc = COURT_POSITION_TASK_COST_BREAKDOWN_BASE
					format = "BASE_VALUE_FORMAT_DECIMALS_PLUS_NEGATIVE"
				}
			}
		}
	}

	employee_modifier = {
		monthly_piety = 1
	}

	scaling_employer_modifiers = {
		terrible = {
		 	monthly_piety = 0.1
			#ADDED
			county_opinion_add = 2
		}
		poor = {
		 	monthly_piety = 0.25
			#ADDED
			county_opinion_add = 4
		}
		average = {
		 	monthly_piety = 0.5
			#ADDED
			county_opinion_add = 6
		}
		good = {
		 	monthly_piety = 0.8
			#ADDED
			county_opinion_add = 8
		}
		excellent = {
		 	monthly_piety = 1.25
			#ADDED
			county_opinion_add = 10
		}
	}

	is_shown = {  }
	is_valid_showing_failures_only = { }

	on_start = { }
	on_end = { }

	ai_will_do = {
		value = ai_zeal
		if = {
			limit = {
				piety_level <= 2
			}
			add = 25
		}
		#ADDED - since it gives popular opinion now....
		if = {
			limit = { 
				any_held_county = {
					county_opinion < 0
					title_is_a_faction_member = yes
				}
			}
			add = 50
		}
		#ADDED - i heard you need piety
		if = {
			limit = { 
				piety < { value = highest_held_title_tier add = -1 multiply = 250 }
			}
			add = 100
		}
		if = {
			limit = {
				has_treasury = no
				monthly_character_income < monthly_court_position_task_cost_double
			}
			add = -1000
		}
		else_if = {
			limit = {
				has_treasury = yes
				monthly_character_treasury_income < monthly_court_position_task_cost_double
			}
			add = -1000
		}
	}
}
