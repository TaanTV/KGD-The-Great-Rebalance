
add_heresiarch_trait_effect = { # Flag is need to prevent trait being removed when gaining it close to conversion
	add_character_flag = {
		flag = new_heresiarch
		days = 3
	}
	if = {
		limit = {
			NOT = { has_trait = heresiarch }
		}
		add_trait = heresiarch
		#ADDED 
		additional_heresiarch_effects = yes
	}
}
