
check_and_add_legitimacy_for_renown_differences = {
	scope:secondary_$SCOPE_TO$ = {
		if = {
			limit = {
				is_valid_for_legitimacy_change = yes
				is_lowborn = no
				#this = scope:secondary_$SCOPE_TO$ # We check that you're marrying them yourself 
				scope:secondary_$SCOPE_FROM$ = { is_lowborn = no }
				dynasty.dynasty_prestige_level < scope:secondary_$SCOPE_FROM$.dynasty.dynasty_prestige_level
				NAND = {		#only once per character.
					has_variable_list = got_legitimacy_for_marrying_those
					is_target_in_variable_list = {
						name = got_legitimacy_for_marrying_those
						target = scope:secondary_$SCOPE_FROM$
					}
				}
			}
			if = {
				limit = {
					OR = {
						NOT = { exists = scope:secondary_actor }
						NOT = { exists = scope:secondary_recipient }
						scope:secondary_actor.age < 16
						scope:secondary_recipient.age < 16
					}
				}
				#This is here to stop it from double-counting
				show_as_tooltip = {
					add_legitimacy = {
						value = scope:secondary_$SCOPE_FROM$.dynasty.dynasty_prestige_level
						subtract = dynasty.dynasty_prestige_level
						multiply = 10
						if = {
							limit = {
								OR = {
									NOT = { exists = primary_spouse }
									primary_spouse ?= scope:secondary_$SCOPE_FROM$
								}
							}
							multiply = 3
						}
					}
				}
			}
			else = {
				add_legitimacy = {
					value = scope:secondary_$SCOPE_FROM$.dynasty.dynasty_prestige_level
					subtract = dynasty.dynasty_prestige_level
					multiply = 10
					if = {
						limit = {
							OR = {
								NOT = { exists = primary_spouse }
								primary_spouse ?= scope:secondary_$SCOPE_FROM$
							}
						}
						multiply = 3
					}
				}	
				add_to_variable_list = {
					name = got_legitimacy_for_marrying_those
					target = scope:secondary_$SCOPE_FROM$
				}
			}
		}
