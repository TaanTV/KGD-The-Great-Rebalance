add_random_yurt_external_building_effect = {
	random_list = {
		1 = {
			trigger = {
				NOT = { has_domicile_building_or_higher = herd_welfare_yurt_01 }
			}
			add_domicile_building = herd_welfare_yurt_01
		}
		10 = {

				NOT = { has_domicile_building_or_higher = character_warfare_yurt_01 }
