# THIS = mercenary_company_leader, PREV = mercenary_company. ROOT does not exist.
mercenary_castles_pulse_effect = {
	save_scope_as = rich_mercenary_leader
	if = {
		limit = {
			primary_title.title_capital_county = {
				NOT = { has_variable = rich_mercenary_no_repeats }
				culture = scope:rich_mercenary_leader.culture
				faith = scope:rich_mercenary_leader.faith
				NOT = {
					title_province = {
						OR = {
							has_holding_type = tribal_holding
							has_holding_type = nomad_holding
							has_holding_type = herder_holding
						}
					}
				}
				NOT = {
					holder = { has_character_flag = got_bkt_misc_events_0010 }
				}
				any_county_province = {
					has_holding = no
					has_ongoing_construction = no 
					barony = { barony_cannot_construct_holding = no }
				}
				OR = {
					NOT = {
						any_county_province = {
							has_holding_type = castle_holding
						}
					}
					AND = {
						any_county_province = {
							has_holding_type = city_holding
						}
						any_county_province = {
							has_holding_type = church_holding
						}
						any_county_province = {
							count < 2		#Dont spam 
							has_holding_type = castle_holding
						}
					}
				}
			}
		}
		primary_title.title_capital_county = { save_scope_as = chosen_county }
	}
	if = {
		limit = { NOT = { exists = scope:chosen_county } }
		scope:rich_mercenary_leader.culture = {
			random_culture_county = {
				limit = {
					NOT = { has_variable = rich_mercenary_no_repeats }
					OR = {
						faith = scope:rich_mercenary_leader.faith
						faith = {
							faith_hostility_level = { 
								target = scope:rich_mercenary_leader.faith 
								value <= faith_astray_level
							}
						}
					}
					NOT = {
						title_province = {
							OR = {
								has_holding_type = tribal_holding
								has_holding_type = nomad_holding
								has_holding_type = herder_holding
							}
						}
					}
					NOT = {
						holder = { has_character_flag = got_bkt_misc_events_0010 }
					}
					any_county_province = {
						has_holding = no
						has_ongoing_construction = no 
						barony = { barony_cannot_construct_holding = no }
					}
					OR = {
						NOT = {
							any_county_province = {
								has_holding_type = castle_holding
							}
						}
						AND = {
							any_county_province = {
								has_holding_type = city_holding
							}
							any_county_province = {
								has_holding_type = church_holding
							}
							any_county_province = {
								count < 2		#Dont spam 
								has_holding_type = castle_holding
							}
						}
					}
				}
				save_scope_as = chosen_county
			}
		}
	}
	if = {
		limit = {
			exists = scope:chosen_county 
		}
		scope:chosen_county = {
			set_variable = {
				name = rich_mercenary_no_repeats
				days = 1
			}
			holder = {
				trigger_event = {
					id = bkt_misc_events.0010
					days = { 6 30 }
				}
			}
		}
	}
}

fired_from_council_position_prestige_loss_effect = {
	if = {
		limit = {
			exists = root.liege_or_court_owner
			years_from_game_start >= 1
		}
		liege_or_court_owner = {
			if = {
				limit = { is_ai = no }
				if = {
					limit = { highest_held_title_tier <= tier_duchy }
					add_prestige = { value = medium_prestige_loss multiply = 1.5 }
				}
				else = {
					add_prestige = major_prestige_loss
				}
			}
			else = {		#lets try actually punishing AI a bit for this
				add_prestige = minor_prestige_loss
			}
		}
	}
}

#REQUIRES: THIS being the councillor and scope:temp_liege_scope being saved
recalc_vassal_stance_council_bonuses = {
	#Actually, exclude counts from this.
	if = {
		limit = { 
			highest_held_title_tier >= tier_duchy 
			any_vassal = { exists = this }		#sometimes you just dont have vassals somehow.
		}
		save_temporary_scope_as = temp_liege_scope
		clear_vassal_stance_council_bonuses = yes
		every_councillor = {
			limit = { is_ruler = yes }
			switch = {
				trigger = has_vassal_stance
				courtly = { destribute_vassal_stance_council_bonuses = { STANCE = courtly } }
				glory_hound = { destribute_vassal_stance_council_bonuses = { STANCE = glory_hound } }
				parochial = { destribute_vassal_stance_council_bonuses = { STANCE = parochial } }
				zealot = { destribute_vassal_stance_council_bonuses = { STANCE = zealot } }
				minority = { destribute_vassal_stance_council_bonuses = { STANCE = minority } }
				barons_and_minor_landholders = { destribute_vassal_stance_council_bonuses = { STANCE = barons_and_minor_landholders } }
				belligerent = { destribute_vassal_stance_council_bonuses = { STANCE = belligerent } }
			}
		}
	}
}

clear_vassal_stance_council_bonuses = {
	every_vassal = {
		remove_opinion = {
			modifier = same_stance_in_council
			target = prev
		}
	}
}

destribute_vassal_stance_council_bonuses = { 
	scope:temp_liege_scope = {
		every_vassal = {
			limit = {
				has_vassal_stance = $STANCE$
				is_councillor = no
			}
			add_opinion = {
				modifier = same_stance_in_council
				target = scope:temp_liege_scope
			}
		}
	}	
}

#character scope, requires scope:landed_title
get_or_set_year_of_last_serious_defensive_war = {
	random_held_title = {
		limit = {
			tier >= tier_empire
			NOT = { this = scope:landed_title }
			has_variable = year_of_last_serious_defensive_war
		}
		save_scope_as = title_to_copy_variable_from
	}
	if = {
		limit = {
			exists = scope:title_to_copy_variable_from
		}
		scope:landed_title = {
			set_variable = {
				name = year_of_last_serious_defensive_war
				value = scope:title_to_copy_variable_from.var:year_of_last_serious_defensive_war
			}
		}
	}
	else = {
		scope:landed_title = {
			set_variable = {
				name = year_of_last_serious_defensive_war
				value = current_year
			}
		}
	}
}
