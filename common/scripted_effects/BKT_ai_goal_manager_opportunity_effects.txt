ai_goal_manager_opportunity_pulse_war_effect = {
	save_scope_as = story_owner_scope
	random = {
		chance = 75
		modifier = {
			add = {
				value = ai_greed 
				add = ai_boldness
				add = ai_vengefulness
				subtract = ai_compassion
				subtract = ai_honor
				#All of the above is a value between -500 and +500 
				multiply = 0.05
				min = -25
				max = 25
			}
		}
		modifier = {	#Special - opinion of liege reduces the chance
			factor = {
				value = 75
				subtract = "opinion(liege)"
				max = 100
				min = 0
				multiply = 0.01
			}
			top_liege != this
		}
		modifier = {
			government_allows = administrative
			top_liege != this
			factor = 0.5
		}
		save_temporary_scope_as = can_wage_war
	}
	random = {
		chance = 33
		modifier = {
			government_allows = administrative
			add = -23
		}
		set_variable = {
			name = will_attack_tribals
			days = 1
		}
	}
	if = {
		limit = {
			exists = scope:can_wage_war
			has_personality_benevolent_trigger = yes
		}
		ai_start_best_war = {
			cb = {
				claim_cb
				de_jure_cb
			}
			is_valid = {
				save_temporary_scope_as = story_owner_scope
				#scope:has_hostage = no
				scope:target_character = {
					story_owner_scope_can_attack_this = yes
					current_military_power_this_plus_allies_value < scope:story_owner_scope.current_military_power_this_plus_allies_value
					OR = {
						scope:score >= 10000		#prev 3000
						AND = {
							NOT = { any_ally = { military_power >= scope:story_owner_scope.a_third_of_current_military_power } } 
							scope:story_owner_scope.a_third_of_current_military_power >= military_power
							scope:score > 2000
						}
					}
				}
				OR = {
					scope:target_character = { NOT = { government_has_flag = government_is_tribal } }
					government_has_flag = government_is_tribal
					has_variable = will_attack_tribals
				}
			}
			on_success = {
				debug_log = "War declared through ai opportunity pulse!"
				debug_log_scopes = yes
				debug_log_date = yes
			}
		}
	}
	else_if = {
		limit = {
			exists = scope:can_wage_war
		}
		ai_start_best_war = {
			cb = {
				claim_cb
				religious_war
				imperial_reconquest_cb
				invasion_war
				clan_invasion_war
				tribal_subjugation_cb
				de_jure_cb
			}
			is_valid = {
				save_temporary_scope_as = story_owner_scope
				#scope:has_hostage = no
				current_military_power_this_plus_allies_value < scope:story_owner_scope.current_military_power_this_plus_allies_value
				scope:target_character = {
					story_owner_scope_can_attack_this = yes
					OR = {
						scope:score >= 10000		#prev 3000
						AND = {
							NOT = { any_ally = { military_power >= scope:story_owner_scope.a_third_of_current_military_power } } 
							scope:story_owner_scope.a_third_of_current_military_power >= military_power
							scope:score > 2000
						}
					}
				}
				OR = {
					scope:target_character = { NOT = { government_has_flag = government_is_tribal } }
					government_has_flag = government_is_tribal
					has_variable = will_attack_tribals
				}
			}
			on_success = {
				debug_log = "War declared through ai opportunity pulse!"
				debug_log_scopes = yes
				debug_log_date = yes
			}
		}
	}
}
