### Adopt [ROOT.Char.Custom('GetTribalReformGovernment')|U] Ways ###
convert_to_feudalism_from_administrative_decision = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_realm.dds"
	}
	decision_group_type = major
	desc = convert_to_feudalism_from_administrative_decision_desc

	ai_check_interval_by_tier = {
		barony = 0
		county = 36
		duchy = 36
		kingdom = 36
		empire = 36
		hegemony = 0
	}
	is_shown = {
		trigger_if = {
			limit = { has_title = title:h_china }
			NOT = { government_has_flag = government_is_celestial }
		}
		government_allows = administrative
		tgp_is_ceremonial_liege_trigger = no
		tgp_is_ceremonial_regent_trigger = no
		tgp_is_any_minister = no
		# Ritsuryo gov has its own decisions
		NOT = { government_has_flag = government_is_japan_administrative }
	}

	is_valid = {
		is_landed = yes
		trigger_if = {
			limit = {
				top_liege != this
			}
			top_liege ?= {
				trigger_if = {
					limit = {
						has_realm_law_in_group = imperial_bureaucracy
					}
					has_realm_law = imperial_bureaucracy_0 
				}
				trigger_if = {
					limit = {
						has_realm_law_in_group = celestial_bureaucracy
					}
					has_realm_law = celestial_bureaucracy_0 
				}
				trigger_if = {
					limit = {
						has_realm_law_in_group = meritocratic_bureaucracy
					}
					has_realm_law = meritocratic_bureaucracy_0 
				}
				trigger_if = {
					limit = {
						has_realm_law_in_group = japanese_bureaucracy
					}
					has_realm_law = japanese_bureaucracy_0 
				}
			}
			custom_tooltip = {
				text = governorship_is_on_the_border_or_borders_feudal_clan_vassal_tt	#prev governorship_is_on_the_border_tt
				#ADDED - an OR, might save on performance to check contract first as it is only valid if the below stuff is valid anyways
				OR = {
					vassal_contract_has_flag = admin_theme_frontier
					any_sub_realm_county = {
						any_neighboring_county = {
							#holder.top_liege != root.top_liege
							#ADDED - be on the border or border someone who's already feudal
							holder ?= {
								OR = {
									top_liege != root.top_liege
									AND = {
										highest_held_title_tier = root.highest_held_title_tier
										OR = { 
											government_has_flag = government_is_feudal
											government_has_flag = government_is_clan
										}
									}
									any_liege_or_above = {
										highest_held_title_tier = root.highest_held_title_tier
										OR = { 
											government_has_flag = government_is_feudal
											government_has_flag = government_is_clan
										}
									}
								}
							}
						}
					}
				}
			}
			#custom_tooltip = {
			#	text = you_have_governorship_rights_tt
			#	primary_title.var:petition_house_rights ?= house
			#}
			#ADDED - bypass to house rights coz AI cant have them
			OR = {
				custom_tooltip = {
					text = you_have_governorship_rights_tt
					primary_title.var:petition_house_rights ?= house
				}
				AND = {
					top_liege ?= { 
						is_in_civil_war = yes 
						custom_description = {
							liege = { debt_level >= 1 } 
							subject = liege
							object = root
							text = subject_is_in_debt_tt
						}
					}
					custom_tooltip = {
						text = your_house_held_primary_title_for_a_generation_tt
						primary_title.previous_holder ?= {
							house ?= root.house
						}
					}
				}
					
			}
			#prestige_level >= 3
			#influence_level >= 3
			trigger_if = {
				limit = {
					primary_title = {
						any_this_title_or_de_jure_above = {
							holder = root.top_liege
						}
					}
				}
				prestige_level >= 3
				influence_level >= 3
			}
			trigger_else = {
				prestige_level >= 2
				influence_level >= 2
			}
		}
		trigger_else = {
			prestige_level >= { # Depends on tier
				value = 3		#prev 2
				if = {
					limit = {
						primary_title.tier >= tier_empire
					}
					add = 1
				}
			}
			trigger_if = {
				limit = {
					has_realm_law_in_group = imperial_bureaucracy
				}
				has_realm_law = imperial_bureaucracy_3
			}
			trigger_if = {
				limit = {
					has_realm_law_in_group = celestial_bureaucracy
				}
				has_realm_law = celestial_bureaucracy_3
			}
