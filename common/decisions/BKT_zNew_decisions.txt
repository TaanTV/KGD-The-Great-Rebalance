usurp_caliph_title_decision = {
	picture = { reference = "gfx/interface/illustrations/decisions/decision_realm.dds" }

	title = usurp_caliph_title_decision
	desc = usurp_caliph_title_decision_desc
	selection_tooltip = usurp_caliph_title_decision_tooltip
	decision_group_type = major

	ai_check_interval_by_tier = {
		barony = 0
		county = 0
		duchy = 0
		kingdom = 36
		empire = 36
		hegemony = 36
	}

	is_shown = {
		is_landed_or_landless_administrative = yes
		exists = dynasty
		religion = religion:islam_religion
		exists = faith.religious_head
		NOT = { this = faith.religious_head }
		highest_held_title_tier >= tier_kingdom
		OR = {
			highest_held_title_tier >= tier_empire
			faith.religious_head = {
				is_lowborn = no
				is_independent_ruler = no
				any_liege_or_above = { this = root }
			}
		}
		# And filter out anyone who'd make this unreliable.
		OR = {
			government_has_flag = government_is_tribal
			government_has_flag = government_is_feudal
			government_has_flag = government_is_clan
			government_allows = administrative
		}
	}

	is_valid = {
		# Must be independent.
		is_independent_ruler = yes
		# Not during the Persian struggle
		#trigger_if = { # Done this way to hide it for Spanish muslims and the like
		#	limit = {
		#		any_character_struggle = {
		#			involvement = involved
		#			is_struggle_type = persian_struggle
		#		}
		#	}
		#	custom_tooltip = {
		#		text = must_not_be_persia_involved_tt
		#		NOT = {
		#			any_character_struggle = {
		#				involvement = involved
		#				is_struggle_type = persian_struggle
		#			}
		#		}
		#	}
		#}
		trigger_if = {
			limit = { faith.religious_head = { is_vassal_of = root } }
			faith.religious_head = { is_vassal_of = root }
		}
		trigger_else = {
			faith.religious_head = { is_vassal_or_below_of = root }
		}
		OR = {
			dynasty = faith.religious_head.dynasty
			has_trait = sayyid
		}
	}

	is_valid_showing_failures_only = {
		is_capable_adult = yes
		is_imprisoned = no
		is_independent_ruler = yes
		is_at_war = no
	}

	effect = {
		faith.religious_head_title = {
			holder = { save_scope_as = previous_caliph }
		}
		scope:previous_caliph = {			
			send_interface_toast = {
				type = msg_lost_single_title
				left_icon = root
				create_title_and_vassal_change = {
					type = usurped
					save_scope_as = change
					add_claim_on_loss = yes
				}
				faith.religious_head_title = {
					change_title_holder = {
						holder = root
						change = scope:change
					}
				}
				resolve_title_and_vassal_change = scope:change
				add_opinion = {
					modifier = usurped_title
					target = root
				}
			} 
		}
	}
	
	cost = {
		piety = {
			if = {
				limit = { NOT = { dynasty = faith.religious_head.dynasty } }
				add = root.medium_piety_value
			}
		}
	}
	
	ai_potential = {
		is_landed_or_landless_administrative = yes
		highest_held_title_tier >= tier_kingdom
		exists = dynasty
		religion = religion:islam_religion
		exists = faith.religious_head
		faith.religious_head = {
			is_lowborn = no
			is_independent_ruler = no
		}
		OR = {
			dynasty = faith.religious_head.dynasty
			AND = {
				has_trait = sayyid
				piety_level > 1
			}
		}
	}

	ai_will_do = {
		base = 100
	}
}
