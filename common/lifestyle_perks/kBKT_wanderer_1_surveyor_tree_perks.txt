
local_inspection_perk = {
	lifestyle = wanderer_lifestyle
	tree = surveyor
	position = { 2 1 }
	icon = node_wanderer
	
	parent = mustering_the_troops_perk

	government_character_modifier = {
		flag = government_is_settled
		domain_tax_mult = 0.10
		barons_and_minor_landholders_tax_contribution_mult = 0.05	#prev 0.1
	}

	name = {
		first_valid = {
			triggered_desc = {
				trigger = {
					exists = this
					NOT = { government_has_flag = government_is_settled }
				}
				desc = local_inspection_perk_adventurer_name
			}
			desc = local_inspection_perk_name
		}
	}
	
	effect = {
		if = {
			limit = {
				government_has_flag = government_is_settled
			}
			custom_description_no_bullet = {
				text = local_inspection_perk_effect
			}
		}
		else_if = {
			limit = {
				government_has_flag = government_is_landless_adventurer
			}
			custom_description_no_bullet = {
				text = local_inspection_perk_2_effect
			}
			hidden_effect = {
				landless_inspector_upgrade_terrain_tier_effect = yes
			}
		}
		else_if = {
			limit = {
				government_has_flag = government_is_nomadic
			}
			custom_description_no_bullet = {
				text = local_inspection_perk_3_effect
			}
			hidden_effect = {
				landless_inspector_upgrade_terrain_tier_effect = yes
			}
		}
	}
}
