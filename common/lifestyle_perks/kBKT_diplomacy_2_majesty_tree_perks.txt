benevolent_intent_perk = {
	lifestyle = diplomacy_lifestyle
	tree = majesty
	
	position = { 0 0 }
	icon = node_diplomacy
	
	auto_selection_weight = {
		value = 11
		if = {
			limit = {
				has_education_diplomacy_trigger = yes
			}
			add = 1989
		}
		if = {
			limit = {
				has_focus = diplomacy_majesty_focus
			}
			multiply = 5
		}
		if = {
			limit = {
				can_start_new_lifestyle_tree_trigger = no
				NOT = { has_perk = firm_hand_perk }
			}
			multiply = 0
		}
		if = { # Conquerors go for the best stuff for war!
			limit = {
				has_variable = conqueror
			}
			add = 10000
		}
	}

	government_character_modifier = {
		flag = government_is_mandala
		tributary_opinion = 10
	}
	
	effect = {
		if = {
			limit = {
				government_has_flag = government_is_landless_adventurer
			}
			custom_tooltip = {
				text = adventurer_benevolent_intent_perk_effect
			}
		}
	}
	
	character_modifier = {
		sway_scheme_phase_duration_add = massive_scheme_phase_duration_bonus_value
		# Wtf do u mean "medium", sway has 1 phase! Pdx PLEASE.
	}
}


true_ruler_perk = {
	lifestyle = diplomacy_lifestyle
	tree = majesty
	
	position = { 1 2 }
	icon = node_diplomacy
	
	parent = inspiring_rule_perk
	parent = praetorian_guard_perk
	
	name = {
		first_valid = {
			triggered_desc = {
				trigger = { has_government = landless_adventurer_government }
				desc = true_ruler_perk_adventurer_name
			}
			desc = true_ruler_perk_name
		}
	}
	
	government_character_modifier = {
		flag = government_is_landless_adventurer
		invert_check = yes
		monthly_prestige_gain_per_legitimacy_level_add = 0.15
		court_grandeur_baseline_add = 15
	}
	government_character_modifier = {
		flag = government_is_mandala
		monthly_piety_gain_per_legitimacy_level_add = 0.15
	}

	effect = {
		if = {
			limit = {
				government_has_flag = government_is_landless_adventurer
			}
			custom_tooltip = {
				text = adventurer_true_ruler_perk_effect
			}
		}
		else = {
			custom_description_no_bullet = {
				text = true_ruler_perk_effect
			}
		}
		if = {
			limit = {
				government_has_flag = government_is_nomadic
			}
			custom_description_no_bullet = {
				text = true_ruler_nomadic_perk_effect
			}
		}
		if = {
			limit = {
				OR = {
					government_has_flag = government_is_nomadic
					government_has_flag = government_is_tribal
				}
			}
			custom_description_no_bullet = {
				text = true_ruler_confederation_perk_effect
			}
		}
	}
}
