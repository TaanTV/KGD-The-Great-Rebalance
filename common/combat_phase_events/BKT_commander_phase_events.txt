	type = commander

	chance = {
		base = 720		#prev 1000
		#ADDED - you break even with vanilla at 14 martial 
		modifier = {
			martial > 0
			add = {
				value = martial
				multiply = 20

				}
			}
		}
		#ADDED - level of fame impact
		modifier = {
			always = yes
			add = {
				value = prestige_level
				min = 0
				multiply = -40
			}
		}
		modifier = {
			add = 400		#prev 1000
			has_perk = stalwart_leader_perk
			is_ai = no
		}
		modifier = {
			add = 500		#prev 300
			has_perk = stalwart_leader_perk
			is_ai = yes
		}
		modifier = {
			add = 200		#prev 300
			dynasty ?= {
				has_dynasty_perk = warfare_legacy_3
			}
		}
		modifier = { # Difficulty
			add = 1000
			is_ai = no
			has_game_rule = easy_difficulty
		}
		modifier = {
			add = 2000
			is_ai = no
			has_game_rule = very_easy_difficulty
		}
		modifier = {
			add = 2000
			ai_should_get_extreme_conqueror_bonuses = yes
		}
		#ADDED - conquerors are a bit easier to kill
		modifier = {
			add = -150	
			ai_has_conqueror_personality = yes
		}
		#ADDED - randos shouldn't die 24/7
		modifier = {
			factor = 1.25
			is_ruler = no
		}
	}
}

commander_wounded = {
	type = commander

	is_valid = {
		exists = root
		has_trait_rank = {
			trait = wounded
			rank <= 3
		}
	}

	chance = {
		base = 30		#prev 25
		modifier = {
			factor = { # Prowess directly reduces chance of injury/death
				value = 45		#prev 40 # Value at which 'immortality' happens
				subtract = prowess
				divide = 36		#prev 40
				#min = 0.1 # Everyone dies.
				#ADDED
				# Explaination: if you have less than 18 prowess the chance is HIGHER.
				# Because (45-18)/36 + 0.25 = 0.75 + 0.25 = 1
				# At 0 prowess you get x1.5 chance and at 45 you get the min x0.25
				min = 0
				add = 0.25	# Everyone dies indeed.
			}
		}
		modifier = { # Reduce chance when outnumbering the enemy
			factor = {
				value = scope:combat_side.enemy_side.side_strength
				divide = scope:combat_side.side_strength
				multiply = 1.4 # Outnumbering 2-1 reduces chance by 30%
				max = 1
				# ADDED
				min = 0.25
			}
			scope:combat_side.side_strength > scope:combat_side.enemy_side.side_strength
		}
		#ADDED - players are nasty.
		#modifier = {
		#	is_ai = no
		#	add = {
		#		value = martial
		#		multiply = 0.5
		#	}
		#}
		modifier = {
			factor = 1.3		#prev 1.1
			faith = {
				has_doctrine_parameter = death_is_glory
			}
		}
		modifier = {
			factor = 2
			has_trait = brave
		}
		modifier = {
			factor = 0.5
			has_trait = craven
		}
		#ADDED - same for paranoid
		modifier = {
			factor = 0.5
			has_trait = paranoid
		}
	}

	effect = {
		save_scope_as = commander
		# Trigger the battle event indicating we have been Wounded.
		if = {
			limit = {
				scope:combat_side.enemy_side = {
					any_side_knight = {
						this.prowess >= scope:commander.prowess_for_combat_event_opponent
					}
				}
			}
			scope:combat_side.enemy_side = {
				random_side_knight = {
					limit = {
						this.prowess >= scope:commander.prowess_for_combat_event_opponent
					}
					save_scope_as = enemy_knight
					add_prestige = knight_prestige_gain_on_wound_inverse
					knight_increase_prowess_chance_effect = yes
				}
			}
			# Trigger the battle event where we are wounded by the enemy Knight.
			scope:combat_side = {
				battle_event = {
					key = "commander_wounded_by_enemy"
					left_portrait = scope:commander
					right_portrait = scope:enemy_knight
					type = wound
				}
			}
			# Send the toast telling the player if they have been wounded.
			send_interface_toast = {
				type = event_toast_effect_bad
				title = commander_wounded_by_enemy_interface_friendly_player
				left_icon = scope:commander
				right_icon = scope:enemy_knight

				# Inflict a wound on the commander.
				increase_wounds_effect = { REASON = fight }
			}
		}
		else = {
			# Trigger the battle event where we are wounded by a nameless soldier.
			scope:combat_side = {
				battle_event = {
					key = "commander_wounded_no_enemy"
					left_portrait = scope:commander
					type = wound
				}
			}
			# Send the toast telling the player if they have been wounded.
			send_interface_toast = {
				type = event_toast_effect_bad
				title = commander_wounded_no_enemy_interface_friendly_player
				left_icon = scope:commander

				# Inflict a wound on the commander.
				increase_wounds_effect = { REASON = fight }
			}
		}
	}
}

		OR = {
			#...they are already wounded...
			has_trait_rank = {
				trait = wounded
				rank >= 1
			}
			#...or they are absolutely terrible at battle to begin with!
			AND = {
				martial <= low_skill_rating
				prowess <= low_skill_rating
			}
			#ADDED - ...or they are famous, big target!
			AND = {
				prestige_level >= high_prestige_level
				martial < extremely_high_skill_rating
			}
			#ADDED - ...or they are very famous, even bigger target!
			AND = {
				prestige_level >= very_high_prestige_level
				martial < 35
			}
			#ADDED - ...or they are in debt!
			AND = {
				highest_held_title_tier >= tier_county
				debt_level < -2
			}
		}

			AND = {
				is_ai = no
