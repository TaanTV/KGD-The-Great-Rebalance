		difficulty ai
	}
	
	default = roleplay_difficulty

	bkt_normal_difficulty = {
		flag = GG_can_change_rule
	}

		apply_modifier = ai:roleplay_difficulty_ai
		flag = GG_can_change_rule
	}

		apply_modifier = player:roleplay_difficulty_player

	}
