diplomacy_adventurer_focus = {
	lifestyle = diplomacy_lifestyle
	
	is_shown = {
		government_has_flag = government_is_landless_adventurer
	}
	is_valid = {
		government_has_flag = government_is_landless_adventurer
	}
	
	modifier = {
		general_opinion = 5						#10
		character_travel_speed_mult = 0.25		#0.3
		diplomacy_scheme_phase_duration = -10 	#-15
		#diplomacy = 1
	}

	desc = {
		desc = diplomacy_adventurer_focus_desc
		desc = line_break
	}

	auto_selection_weight = {
		value = 5000
		if = {
			limit = {
				has_education_diplomacy_trigger = yes
			}
			add = 50000
		}
		if = {		#ADDED
			limit = {
				has_completed_target_lifestyle_trigger = { TARGET = diplomacy }
			}
			multiply = 0
		}
	}
}
