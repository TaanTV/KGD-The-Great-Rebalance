ai_tier_1_economical_building_preference_modifier = {
	modifier = { # Economy boom AI's like economy buildings
		add = 5
		scope:holder = {
			OR = {
				ai_has_economical_boom_personality = yes
				has_character_flag = ai_desire_economy_buildings
			}
		}
	}
	modifier = {
		add = -10
		free_building_slots <= 2
		scope:holder == scope:county.holder
		has_any_main_economic_building = yes
		NOT = { has_any_basic_military_building = yes }
	}
	modifier = {
		add = -2000
		free_building_slots <= 1
		scope:holder == scope:county.holder
		has_any_main_economic_building = yes
		NOT = { has_any_basic_military_building = yes }
	}
}
