# called for the newborn child
# scope:child is the new born character
# scope:mother is the pregnant mother
# scope:father is the family father (not the biological)
# scope:real_father is the biological father
# scope:is_bastard, true if a known bastard
on_birth_child = {
	on_actions = {
		delay = { days = 1 }
		bkt_on_birth_child_nomadic_philosophy
		bkt_on_birth_child_seclusion
	}
}

bkt_on_birth_child_nomadic_philosophy = {
	trigger = {
		has_trait = nomadic_philosophy
	}
	effect = {
		# MPO: Nomadic Philosophy Inheritance
		if = {
			limit = {
				OR = {
					scope:mother ?= {
						matrilinear_marriage = yes
						NOT = { has_trait = nomadic_philosophy }
					}
					scope:father ?= {
						patrilinear_marriage = yes
						NOT = { has_trait = nomadic_philosophy }
					}
				}
			}
			remove_trait = nomadic_philosophy
		}
	}
}

bkt_on_birth_child_seclusion = {
	trigger = {
		OR = {
			scope:mother ?= { has_character_modifier = isolating_modifier }
			scope:father ?= { has_character_modifier = isolating_modifier }
		}
	}
	effect = {
		liege ?= {
			if = {
				limit = { has_variable_list = isolating_relatives }
				add_to_variable_list = {
					name = isolating_relatives
					target = prev
				}
			}
		}
		add_character_flag = {
			flag = blocked_from_leaving
			years = 5
		}
		add_character_modifier = {
			modifier = isolating_modifier
			years = 5
		}
	}
}
